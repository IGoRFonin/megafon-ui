import { DemoNotificationWrapper, wrapperStyle } from './Notification.docz';
import Attention from 'icons/System/16/Attention_16.svg';
import Rating1 from 'icons/System/24/Rating_1_24.svg';
import Checked from 'icons/System/16/Checked_16.svg';
import Rating3 from 'icons/System/24/Rating_3_24.svg';
import Info from 'icons/System/16/Info_16.svg';

```javascript collapse=Код DemoNotificationWrapper
export const DemoNotificationWrapper: React.FC<Props> = ({ initialCloseClickAmount = 0, initialLinkClickAmount = 0, children }) => {
    const [closeClickAmount, setCloseClickAmount] = useState(initialCloseClickAmount);
    const [linkClickAmount, setLinkClickAmount] = useState(initialLinkClickAmount);

    const handleCloseButtonClick = (): void => {
        setCloseClickAmount(closeClickAmount + 1)
    }

    const handleLinkClick = (): void => {
        setLinkClickAmount(linkClickAmount + 1)
    }

    return (
        <div>
            {children({
                onLinkClick: handleLinkClick,
                onClose: handleCloseButtonClick,
                initialLinkClickAmount: linkClickAmount,
                initialCloseClickAmount: closeClickAmount
            })}
        </div>
    );
};
```
